Pseudocódigo de atualização de pesos¶

update_weights(x,y,local_error)
    self.w <- self.w + x*y*local_error
    self.bias <- self.bias + y*local_error

def signal(value):
    return 2*(T.gt(value,0))-1

class Perceptron(object):
    
    def __init__(self,x_example,activation,initial_value=0.):
        W = np.zeros_like(x_example) + initial_value
        bias = np.float32(initial_value)
        self.W = theano.shared(W,name="W")
        self.x_input = T.fvector(name="x_input")
        self.y_input = T.fscalar(name="y_input")
        self.bias = theano.shared(bias,name="bias")

        ### PREDIO
        self.wx_bias = self.W.dot(self.x_input) + self.bias
        self.y_chapeu = activation(self.wx_bias)
        self.predict = theano.function(
            inputs=[self.x_input],
            outputs=[self.y_chapeu]) 

        ### CUSTO
        error = 1-self.wx_bias*self.y_input
        self.local_error = T.maximum(0,error) # Derivada do theano pro T.maximum  um pouco diferente.
                                     

        ### GRADIENTE E UPDATES
        self.w_grad, self.bias_grad = T.grad(
            self.local_error, [self.W, self.bias]) 
        #Calculo da derivada do erro, baseado em W e bias
        self.new_W = self.W - self.w_grad
        self.new_bias = self.bias - self.bias_grad

        ### FUNO FIT
        self.fit_theano = theano.function(
            inputs=[self.x_input, self.y_input], 
            outputs=[self.y_chapeu,self.local_error],
            updates=[
                     (self.W,self.new_W),
                     (self.bias,self.new_bias)
                    ]
            )
    def fit(self,x,y,_max=1000):
        error = 1
        count = 0
        while error > 0 and count < _max:
            error = 0
            count += 1
            for _x,_y in zip(x,y):
                y_chapeu, local_error = self.fit_theano(_x,_y)
                error += local_error[0]
        return error == 0, count

O xor impossvel com os dados atuais pois temos uma impossibilidade lgica

$x = [0,0]$ $y = -1$. Logo $bias \leq 0$
$x = [1,0]$ $y = 1$. Logo $-bias \leq W[0] $
$x = [0,1]$ $y = 1$. Logo $-bias \leq W[1] $
$x = [1,1]$ $y = -1$. Logo $W[0] + W[1] \leq -bias$
$bias \leq 0 \implies |bias| = - bias$
Ento $$ |bias| \leq W[0] \implies W[0] \geq 0 $$ $$ |bias| \leq W[1] \implies W[1] \geq 0$$ $$ W[0] + W[1] \leq |bias|$$ logo $$ W[0] \geq |bias| \land W[1] \geq |bias| \land W[0] + W[1] \leq |bias| $$

Impossvel

Solução?¶

Para resolvermos esse problema vamos adicionar um valor para a entrada que o valor do and entre os dois valores de entrada. Feito isso teremos:

$x = [0,0,0]$ $y = -1$. Logo $bias \leq 0$
$x = [1,0,0]$ $y = 1$. Logo $-bias \leq W[0] $
$x = [0,1,0]$ $y = 1$. Logo $-bias \leq W[1] $
$x = [1,1,1]$ $y = -1$. Logo $W[0] + W[1] + W[2] \leq -bias$
$bias \leq 0 \implies |bias| = - bias$
Ento $$ |bias| \leq W[0] \implies W[0] \geq 0 $$ $$ |bias| \leq W[1] \implies W[1] \geq 0$$ $$ W[0] + W[1] + W[2] \leq |bias|$$ logo $$ W[0] \geq |bias| \land W[1] \geq |bias| \land W[2] \leq |bias| - |W[0]| - |W[1]| $$

No impossvel
